select al.nombreAlumno, em.nombreEmpresa, resp.nombreResponsable, tutor.nombreTutor from fct_alumno as fct inner join alumno as al on al.dniAlumno= fct.dniAlumno 
	inner join cursos as c on c.id=al.curso inner join empresas as em on em.id= fct.idEmpresa 
    inner join responsable as resp on resp.dniResponsable= fct.dniResponsable 
    inner join 
    	tutorescolar as tutor  on  tutor.dniTutorEscolar = fct.dniTutor
where c.id="1";

	inner join cursos as c on c.id=al.curso inner join empresas as em on em.id= fct.idEmpresa 
    inner join responsable as resp on resp.dniResponsable= fct.dniResponsable 
    inner join 
    	tutorescolar as tutor  on  tutor.dniTutorEscolar = fct.dniTutor
